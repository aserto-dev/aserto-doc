# users.conf

```
# use the file input to monitor the directory where we are dropping the files
input {
  file {
    path => ["~/files/decision_logs/0116e83a-7e21-11ec-ab5b-00c9e2c2068b/users/*"]
    mode => "read"
    file_completed_action => "log"
    file_completed_log_path => "~/files/decision_logs/0116e83a-7e21-11ec-ab5b-00c9e2c2068b/users-completed.log"

    # user objects are JSON, so use the json codec
    codec => json
  }
}

# indexing map fields is complex, so for simplicity, remove the identities map
# from the object
filter {
  mutate {
    remove_field => [ "identities" ]
  }
}

# write it all to a new index
output {
  elasticsearch {
    hosts => ["localhost:9200"]
    user => "elastic"
    password => "password"
    index => "aserto-users-0116e83a-7e21-11ec-ab5b-00c9e2c2068b"
    document_id => "%{[id]}"
  }
}
```
